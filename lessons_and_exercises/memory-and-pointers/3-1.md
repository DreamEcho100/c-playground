void allocate_double(double **ptr_to_ptr, double value)
{
 double *heap_memory = (double*)malloc(sizeof(double));
 if (heap_memory == NULL)
 {
  fprintf(stderr, "Memory allocation failed!\n");
  exit(1);
 }

 *heap_memory = value;

 *ptr_to_ptr = heap_memory;
}
